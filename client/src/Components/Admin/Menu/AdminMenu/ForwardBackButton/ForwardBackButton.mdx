# Forward Back Button

Updated 20/06/2019

Navigates back and forth between home and to locations.
Displays different icons based on the current pathname.

<Props of={ForwardBackButton} />

<Playground>
  {() => {
    const [path, setPath] = useState('/home/');
  
    return (
      <div>
        <span>Current path: {path}</span>
        <Router>
          <ForwardBackButton
            home="/home/"
            to="/to/"
            onForward={() => {
              setPath('/to/');
            }}
            onBack={() => {
              setPath('/home/');
            }}
            pathname={location.pathname}
            homeIcon={AddIcon}
            toIcon={ArrowBackIcon}
          />
        </Router>
        
      </div>
      )
    }}
  
   
</Playground>
